843 template< typename StateScalarType = double, typename TimeType = double >
844 std::shared_ptr< ephemerides::Ephemeris > createTabulatedEphemerisFromSpice(
845  const std::string& body,
846  const TimeType initialTime,
847  const TimeType endTime,
848  const TimeType timeStep,
849  const std::string& observerName,
850  const std::string& referenceFrameName,
851  std::shared_ptr< interpolators::InterpolatorSettings > interpolatorSettings =
852  std::make_shared< interpolators::LagrangeInterpolatorSettings >( 8 ) )
853 {
854  using namespace interpolators;
855 
856  std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > timeHistoryOfState;
857 
858  // Calculate state from spice at given time intervals and store in timeHistoryOfState.
859  TimeType currentTime = initialTime;
860  while( currentTime < endTime )
861  {
862  timeHistoryOfState[ currentTime ] = spice_interface::getBodyCartesianStateAtEpoch(
863  body, observerName, referenceFrameName, "none", static_cast< double >( currentTime ) ).
864  template cast< StateScalarType >( );
865  currentTime += timeStep;
866  }
867 
868  // Create interpolator.
869  std::shared_ptr< OneDimensionalInterpolator< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > > interpolator =
870  interpolators::createOneDimensionalInterpolator(
871  timeHistoryOfState, interpolatorSettings );
872 
873  // Create ephemeris and return.
874  return std::make_shared< ephemerides::TabulatedCartesianEphemeris< StateScalarType, TimeType > >(
875  interpolator, observerName, referenceFrameName );
876 }

878 template< typename StateScalarType = double, typename TimeType = double >
879 std::shared_ptr< ephemerides::Ephemeris > createTabulatedEphemerisFromTLE(
880  const std::string& body,
881  const TimeType initialTime,
882  const TimeType endTime,
883  const TimeType timeStep,
884  const std::string& observerName,
885  const std::string& referenceFrameName,
886  std::shared_ptr< ephemerides::Tle > tle,
887  std::shared_ptr< interpolators::InterpolatorSettings > interpolatorSettings =
888  std::make_shared< interpolators::LagrangeInterpolatorSettings >( 8 ) )
889 {
890  using namespace interpolators;
891 
892  std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > timeHistoryOfState;
893 
894  // Calculate state from spice at given time intervals and store in timeHistoryOfState.
895  TimeType currentTime = initialTime;
896  while( currentTime < endTime )
897  {
898  timeHistoryOfState[ currentTime ] = spice_interface::getCartesianStateFromTleAtEpoch( static_cast< double >( currentTime ), tle );
899  currentTime += timeStep;
900  }
901 
902  // Create interpolator.
903  std::shared_ptr< OneDimensionalInterpolator< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > > interpolator =
904  interpolators::createOneDimensionalInterpolator(
905  timeHistoryOfState, interpolatorSettings );
906 
907  // Create ephemeris and return.
908  return std::make_shared< ephemerides::TabulatedCartesianEphemeris< StateScalarType, TimeType > >(
909  interpolator, observerName, referenceFrameName );
910 }

930 inline std::shared_ptr< EphemerisSettings > keplerEphemerisFromSpiceSettings(
931  const std::string body,
932  const double epochOfInitialState,
933  const double centralBodyGravitationalParameter,
934  const std::string& referenceFrameOrigin = "SSB",
935  const std::string& referenceFrameOrientation = "ECLIPJ2000",
936  const double rootFinderAbsoluteTolerance =
937  200.0 * std::numeric_limits< double >::epsilon( ),
938  const double rootFinderMaximumNumberOfIterations = 1000.0 )
939 {
940  Eigen::Vector6d initialCartesianState = spice_interface::getBodyCartesianStateAtEpoch(
941  body, referenceFrameOrigin, referenceFrameOrientation, "None", epochOfInitialState );
942  Eigen::Vector6d initialKeplerianState = orbital_element_conversions::convertCartesianToKeplerianElements(
943  initialCartesianState, centralBodyGravitationalParameter );
944  return std::make_shared< KeplerEphemerisSettings >(
945  initialKeplerianState, epochOfInitialState, centralBodyGravitationalParameter,
946  referenceFrameOrigin, referenceFrameOrientation, rootFinderAbsoluteTolerance,
947  rootFinderMaximumNumberOfIterations );
