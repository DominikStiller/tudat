820 class AccelerationLimitedThrottleGuidance: public ThrustInputParameterGuidance
821 {
822 public:
823 
825 
836  AccelerationLimitedThrottleGuidance(
837  const SystemOfBodies& bodies,
838  const std::string nameOfBodyWithGuidance,
839  const std::string nameOfCentralBody,
840  const std::vector< propulsion::ThrustIndependentVariables > independentVariables,
841  const std::shared_ptr< interpolators::Interpolator< double, double > > thrustInterpolator,
842  const double maximumAcceleration ): ThrustInputParameterGuidance( 1, 0, true, 0 ),
843  bodyWithGuidance_( bodies.at( nameOfBodyWithGuidance ) ), thrustInterpolator_( thrustInterpolator ),
844  maximumAcceleration_( maximumAcceleration )
845  {
846  // Split independent variables into environmental/guidance.
847  int numberOfThrottles = 0;
848  std::vector< propulsion::ThrustIndependentVariables > guidanceFreeIndependentVariables;
849  for( unsigned int i = 0; i < independentVariables.size( ); i++ )
850  {
851  if( independentVariables.at( i ) == propulsion::throttle_dependent_thrust )
852  {
853  numberOfThrottles ++;
854  }
855  else if( independentVariables.at( i ) == propulsion::guidance_input_dependent_thrust )
856  {
857  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, guidance input not supported" );
858  }
859  else
860  {
861  guidanceFreeIndependentVariables.push_back( independentVariables.at( i ) );
862  }
863  }
864 
865  // Check input consistency.
866  if( numberOfThrottles > 1 )
867  {
868  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, multiple throttles detected" );
869  }
870 
871  // Create thrust input.
872  if( bodyWithGuidance_->getFlightConditions( ) == nullptr && nameOfCentralBody != "" )
873  {
874  bodyWithGuidance_->setFlightConditions(
875  createAtmosphericFlightConditions( bodyWithGuidance_,
876  bodies.at( nameOfCentralBody ),
877  nameOfBodyWithGuidance,
878  nameOfCentralBody ) );
879  }
880  thrustInputFunctions_ = getPropulsionInputVariables(
881  bodyWithGuidance_, guidanceFreeIndependentVariables );
882 
883  currentThrustInput_.resize( thrustInputFunctions_.size( ) );
884  }
885 
887 
891  void updateGuidanceParameters( )
892  {
893  // Compute environmental input variables for thrust.
894  for( unsigned int i = 0; i < thrustInputFunctions_.size( ); i++ )
895  {
896  currentThrustInput_[ i ] = thrustInputFunctions_.at( i )( );
897  }
898 
899  // Compute maximum thrust
900  double currentThrust = thrustInterpolator_->interpolate( currentThrustInput_ );
901  double currentMass = bodyWithGuidance_->getBodyMass( );
902 
903  // Set throttle (to < 1 if necessary)
904  if( currentThrust / currentMass < maximumAcceleration_ )
905  {
906  currentThrustGuidanceParameters_[ 0 ] = 1.0;
907  }
908  else
909  {
910  currentThrustGuidanceParameters_[ 0 ] = maximumAcceleration_ * currentMass / currentThrust;
911  }
912  }
913 
914 private:
915 
917  std::shared_ptr< simulation_setup::Body > bodyWithGuidance_;
918 
920  std::shared_ptr< interpolators::Interpolator< double, double > > thrustInterpolator_;
921 
923  double maximumAcceleration_;
924 
926  std::vector< std::function< double( ) > > thrustInputFunctions_;
927 
929  std::vector< double > currentThrustInput_;
930 
931 };
