570 inline std::shared_ptr< simulation_setup::ThrustAccelerationSettings > getLowThrustLegAccelerationSettings(
571  const std::shared_ptr< low_thrust_trajectories::LowThrustLeg > lowThrustLeg,
572  const simulation_setup::SystemOfBodies& bodies,
573  const std::string& bodyToPropagate,
574  const std::function< double( const double ) > specificImpulseFunction,
575  const double lowThrustLegInitialTime )
576 {
577  using namespace low_thrust_trajectories;
578 
579  std::shared_ptr< simulation_setup::Body > vehicle = bodies.at( bodyToPropagate );
580 
581  // Define thrust magnitude function from the shaped trajectory.
582  std::function< double( const double ) > thrustForceMagnitudeFunction;
583  if( lowThrustLeg->getLegModelIsForceBased( ) )
584  {
585  thrustForceMagnitudeFunction =
586  std::bind( &LowThrustLeg::getForceBasedThrustMagnitude, lowThrustLeg,
587  std::placeholders::_1, lowThrustLegInitialTime );
588  }
589  else
590  {
591  std::function< double( const double, const double ) > thrustAccelerationMagnitudeFunction =
592  std::bind( &LowThrustLeg::getAccelerationBasedThrustMagnitude, lowThrustLeg,
593  std::placeholders::_1, lowThrustLegInitialTime, std::placeholders::_2 );
594  std::function< double( ) > bodyMassFunction =
595  std::bind( &Body::getBodyMass, vehicle );
596  thrustForceMagnitudeFunction = [=](const double currentTime ){
597  return thrustAccelerationMagnitudeFunction( currentTime, bodyMassFunction( ) ); };
598  }
599 
600  // Define thrust magnitude settings from thrust magnitude function.
601  std::shared_ptr< simulation_setup::FromFunctionThrustMagnitudeSettings > thrustMagnitudeSettings =
602  std::make_shared< simulation_setup::FromFunctionThrustMagnitudeSettings >(
603  thrustForceMagnitudeFunction, specificImpulseFunction );
604 
605 
606  // Define thrust direction function from the shaped trajectory.
607  std::function< Eigen::Vector3d( const double ) > thrustDirectionFunction =
608  std::bind( &LowThrustLeg::getThrustDirection, lowThrustLeg,
609  std::placeholders::_1, lowThrustLegInitialTime );
610 
611  // Define thrust direction settings from the direction of thrust acceleration retrieved from the shaping method.
612  std::shared_ptr< simulation_setup::CustomThrustDirectionSettings > thrustDirectionSettings =
613  std::make_shared< simulation_setup::CustomThrustDirectionSettings >( thrustDirectionFunction );
614 
615  // Define thrust acceleration settings.
616  std::shared_ptr< simulation_setup::ThrustAccelerationSettings > thrustAccelerationSettings =
617  std::make_shared< simulation_setup::ThrustAccelerationSettings >(
618  thrustDirectionSettings, thrustMagnitudeSettings );
619 
620  return thrustAccelerationSettings;
621 }
